# Bases-de-datos
Para crear la base de datos: <br>
CREATE TABLE `usuario` ( <br>
        `id` INT(10) AUTO_INCREMENT PRIMARY KEY, <br>
  	    `nombre` VARCHAR(20), <br>
        `apellido` VARCHAR(20), <br>
        `email` VARCHAR(20)) <br>
